-- Table: categories
CREATE TABLE categories (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    names NVARCHAR(200) NOT NULL
);

-- Table: articles
CREATE TABLE articles (
    idArticles BIGINT AUTO_INCREMENT PRIMARY KEY,
    title NVARCHAR(200) NOT NULL,
    content TEXT NOT NULL,
    images VARCHAR(500) NOT NULL,
    postedDate DATE NOT NULL,
    categoryId BIGINT NOT NULL,
    home BOOLEAN NOT NULL,
    FOREIGN KEY (categoryId) REFERENCES categories(id)
        ON DELETE NO ACTION
        ON UPDATE CASCADE
);

-- Table: users
CREATE TABLE users (
    userName CHAR(10) PRIMARY KEY,
    pass CHAR(200) NOT NULL,
    fullname NVARCHAR(200) NOT NULL,
    role BOOLEAN DEFAULT 0 NOT NULL,
    email NVARCHAR(200) NOT NULL UNIQUE,
    avatar TEXT
);

-- Table: comments
CREATE TABLE comments (
    idComments BIGINT AUTO_INCREMENT PRIMARY KEY,
    userName CHAR(10) NOT NULL,
    idArticles BIGINT NOT NULL,
    content TEXT NOT NULL,
    dateCmt DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL,
    FOREIGN KEY (userName) REFERENCES users(userName)
        ON DELETE NO ACTION
        ON UPDATE CASCADE,
    FOREIGN KEY (idArticles) REFERENCES articles(idArticles)
        ON DELETE NO ACTION
        ON UPDATE CASCADE
);
